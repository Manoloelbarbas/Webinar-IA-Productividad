import React, { useState } from 'react';
import { Zap, Brain, Code, Database, Sparkles } from 'lucide-react';

export default function App() {
  const [hoveredFeature, setHoveredFeature] = useState(null);

  // Datos para el gráfico de barras horizontales
  const performanceData = [
    { 
      category: 'Razonamiento Autónomo', 
      free: 70, 
      pro: 99, 
      icon: <Brain className="w-5 h-5" />,
      desc: 'Resolución de problemas complejos y creación de agentes que actúan por ti.'
    },
    { 
      category: 'Programación de Sistemas', 
      free: 65, 
      pro: 98, 
      icon: <Code className="w-5 h-5" />,
      desc: 'Arquitectura de software, depuración profunda y creación de apps completas.'
    },
    { 
      category: 'Contexto Masivo (Memoria)', 
      free: 50, 
      pro: 100, 
      icon: <Database className="w-5 h-5" />,
      desc: 'Análisis simultáneo de repositorios gigantes, horas de video y miles de documentos.'
    },
    { 
      category: 'Procesamiento Multimodal', 
      free: 60, 
      pro: 97, 
      icon: <Zap className="w-5 h-5" />,
      desc: 'Comprensión profunda de video de larga duración, audio y gráficos complejos.'
    },
    { 
      category: 'Precisión y Creatividad', 
      free: 80, 
      pro: 99, 
      icon: <Sparkles className="w-5 h-5" />,
      desc: 'Seguimiento milimétrico de instrucciones complejas con mínimas alucinaciones.'
    }
  ];

  // Cálculos para el Gráfico de Radar (Araña)
  const chartSize = 340;
  const chartCenter = chartSize / 2;
  const chartRadius = 110; // Ajustado para dar espacio a las etiquetas
  const numPoints = performanceData.length;

  const getPoint = (value, index) => {
    const angle = (Math.PI / 2) - (2 * Math.PI * index / numPoints);
    const r = (value / 100) * chartRadius;
    return `${chartCenter + r * Math.cos(angle)},${chartCenter - r * Math.sin(angle)}`;
  };

  const proPolygon = performanceData.map((d, i) => getPoint(d.pro, i)).join(' ');
  const freePolygon = performanceData.map((d, i) => getPoint(d.free, i)).join(' ');

  return (
    <div className="min-h-screen bg-slate-50 flex items-center justify-center p-4 md:p-8 font-sans">
      <div className="w-full max-w-2xl bg-white rounded-3xl p-6 md:p-10 shadow-xl shadow-slate-200/50 border border-slate-100 flex flex-col">
        
        <div className="text-center mb-8">
          <h2 className="text-3xl font-bold mb-3 flex items-center justify-center text-slate-900">
            <Zap className="w-8 h-8 text-blue-500 mr-3" />
            Expansión de Capacidades
          </h2>
          <p className="text-slate-500 text-base">
            Visualiza cómo el poder de 3.1 Pro envuelve y supera al modelo gratuito.
          </p>
        </div>
        
        {/* Gráfico de Radar SVG */}
        <div className="relative flex justify-center items-center flex-grow mb-8 mt-4">
          <svg width={chartSize} height={chartSize} className="overflow-visible">
            {/* Círculos de fondo (rejilla) */}
            {[20, 40, 60, 80, 100].map((level) => (
              <circle 
                key={level} 
                cx={chartCenter} 
                cy={chartCenter} 
                r={(level / 100) * chartRadius} 
                fill="none" 
                stroke="#e2e8f0" 
                strokeWidth="1"
                strokeDasharray={level === 100 ? "0" : "4 4"}
              />
            ))}
            
            {/* Ejes */}
            {performanceData.map((_, i) => {
              const [x, y] = getPoint(100, i).split(',');
              return (
                <line 
                  key={`axis-${i}`} 
                  x1={chartCenter} 
                  y1={chartCenter} 
                  x2={x} 
                  y2={y} 
                  stroke="#e2e8f0" 
                  strokeWidth="1" 
                />
              );
            })}

            {/* Polígono Free (3 Flash) */}
            <polygon 
              points={freePolygon} 
              fill="rgba(148, 163, 184, 0.2)" 
              stroke="#94a3b8" 
              strokeWidth="2" 
              className="transition-all duration-1000 ease-out hover:fill-slate-300/40"
            />

            {/* Polígono Pro (3.1 Pro) */}
            <polygon 
              points={proPolygon} 
              fill="url(#proGradient)" 
              stroke="#8b5cf6" 
              strokeWidth="3" 
              className="transition-all duration-1000 ease-out hover:opacity-90 cursor-pointer"
            />

            {/* Puntos y Etiquetas */}
            {performanceData.map((item, i) => {
              const [x, y] = getPoint(125, i).split(','); // Etiquetas más afuera
              const [px, py] = getPoint(item.pro, i).split(','); // Puntos Pro
              const [fx, fy] = getPoint(item.free, i).split(','); // Puntos Free
              
              return (
                <g key={`label-${i}`}>
                  <circle cx={fx} cy={fy} r="4" fill="#94a3b8" />
                  <circle cx={px} cy={py} r="5" fill="#8b5cf6" className="animate-pulse" />
                  
                  <foreignObject 
                    x={x - 60} 
                    y={y - 25} 
                    width="120" 
                    height="80"
                    className="overflow-visible"
                  >
                    <div 
                      className="flex flex-col items-center justify-start text-center cursor-pointer group"
                      onMouseEnter={() => setHoveredFeature(i)}
                      onMouseLeave={() => setHoveredFeature(null)}
                    >
                       <div className={`p-2 rounded-full shadow-sm border transition-all duration-300 ${hoveredFeature === i ? 'bg-purple-50 border-purple-200 text-purple-600 scale-110 shadow-purple-200' : 'bg-white border-slate-100 text-slate-500'}`}>
                         {item.icon}
                       </div>
                       <span className={`text-[10px] font-bold mt-1.5 leading-tight w-28 transition-colors ${hoveredFeature === i ? 'text-purple-700' : 'text-slate-600'}`}>
                         {item.category}
                       </span>
                    </div>
                  </foreignObject>
                </g>
              );
            })}

            <defs>
              <linearGradient id="proGradient" x1="0%" y1="0%" x2="100%" y2="100%">
                <stop offset="0%" stopColor="rgba(139, 92, 246, 0.45)" /> {/* Púrpura */}
                <stop offset="100%" stopColor="rgba(59, 130, 246, 0.25)" /> {/* Azul */}
              </linearGradient>
            </defs>
          </svg>
        </div>

        {/* Leyenda y Descripción interactiva */}
        <div className="bg-slate-50 p-5 rounded-2xl border border-slate-100 min-h-[110px] flex flex-col justify-center transition-all mt-auto shadow-inner">
          {hoveredFeature !== null ? (
            <div className="animate-in fade-in zoom-in duration-200">
              <h4 className="font-bold text-purple-700 flex items-center mb-1 text-sm">
                {performanceData[hoveredFeature].icon}
                <span className="ml-2">{performanceData[hoveredFeature].category}</span>
              </h4>
              <p className="text-slate-600 text-sm leading-relaxed">{performanceData[hoveredFeature].desc}</p>
            </div>
          ) : (
            <div className="flex flex-col items-center text-center space-y-3 opacity-90">
              <div className="flex justify-center space-x-6">
                <div className="flex items-center text-xs font-bold text-slate-500">
                  <div className="w-4 h-4 rounded-md bg-slate-200 border border-slate-300 mr-2"></div>
                  3 Flash (Gratis)
                </div>
                <div className="flex items-center text-xs font-bold text-purple-700">
                  <div className="w-4 h-4 rounded-md bg-gradient-to-br from-purple-400 to-blue-400 border border-purple-300 mr-2"></div>
                  3.1 Pro (Advanced)
                </div>
              </div>
              <p className="text-xs text-slate-400 font-medium">Pasa el ratón sobre los iconos del gráfico para ver detalles.</p>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
